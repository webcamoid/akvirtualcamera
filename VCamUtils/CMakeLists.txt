            src/fraction.cpp

            src/ipcbridge.h
            src/logger.cpp
            src/logger.h
            src/message.cpp
            src/message.h
            src/messageclient.cpp
            src/messageclient.h
            src/messageserver.cpp
            src/messageserver.h
            src/rect.cpp
            src/rect.h
            src/servicemsg.h
            src/settings.cpp
            src/settings.h
            src/sharedmemory.cpp
            src/sharedmemory.h
            src/sockets.cpp
            src/sockets.h
            src/timer.cpp
            src/timer.h
            src/utils.cpp
            src/utils.h
            src/videoadjusts.cpp
            src/videoadjusts.h
            src/videoconverter.cpp
            src/videoconverter.h
            src/videoformat.cpp
            src/videoformat.h
            src/videoformatspec.cpp
            src/videoformatspec.h
            src/videoformattypes.h
            src/videoframe.cpp
            src/videoframe.h
            src/videoframetypes.h)
